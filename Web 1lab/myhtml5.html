D=b2-4ac

2. Алгоритм решения в виде функции на языке Pascal

    program uravnenie;
    var a, b, c, d, x1, x2: real; 
        begin 
        writeln ('Ведите коофицент а'); 
        read (a); 
        writeln ('Ведите коофицент b'); 
        read (b);
        writeln ('Ведите коофицент с');
        read (c);
        d:=(b*b)-4*a*c;
        writeln ('Ваш дискрименант (D) равен ', d);
        if d<0 then writeln ('Корней нет');
        if d>0 then
        d:=sqrt(d);
        x1:=(-b+d)/2*a;
        writeln ('Ваш 1 корень равен ', x1);
        x2:=(-b-d)/2*a;
        writeln ('Ваш 2 корень равен ', x2);
        if d=0 then
        x1:=(-b)/2*a;
        writeln ('Ваш корень равен ', x1);
        end.
